module Piece: sig .. end
type k =
| Pawn
| Knight
| Bishop
| Rook
| Queen
| King 
type c =
| White
| Black 
type w = int 
type l = char * int 
type n_m = l list 
type piece = {
   name : k;
   color : c;
   worth : w;
   mutable location : l;
   mutable next_moves : n_m;
   mutable num_moves : int;
} 
val copy_piece : piece -> piece
val pp_print_loc : char * int -> unit
val pp_type : k -> unit
val pp_piece : piece -> unit
val pp_pl : piece list -> unit
